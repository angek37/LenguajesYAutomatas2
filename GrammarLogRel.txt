<EXPR> ::= <LOG_EXPR>
	| <RELATION_EXPR>
	| <ARITHM_EXPR>
	| id assign Integer
<LOG_EXPR> ::= <LOG_TERM>
<LOG_TERM> ::= <LOG_TERM> and <LOG_FACTOR>
	| <LOG_TERM> or <LOG_FACTOR>
	| <LOG_FACTOR>
<LOG_FACTOR> ::= bool
	| not <LOG_FACTOR>
	| lparen <LOG_EXPR> rparen
	| lparen <RELATION_EXPR> rpaen
<RELATION_EXPR> ::= <ARITHM_FACTOR> greater <ARITHM_FACTOR>
	| <ARITHM_FACTOR> greaterequal <ARITHM_FACTOR>
	| <ARITHM_FACTOR> less <ARITHM_FACTOR>
	| <ARITHM_FACTOR> lessequal <ARITHM_FACTOR>
	| <ARITHM_FACTOR> equal <ARITHM_FACTOR>
	| <ARITHM_FACTOR> notequal <ARITHM_FACTOR>
<ARITHM_EXPR> ::= <ARITHM_TERM> plus <ARITHM_TERM>
	| <ARITHM_TERM> minus <ARITHM_TERM>
<ARITHM_TERM> ::= <ARITHM_FACTOR>
	| <ARITHM_FACTOR> times <ARITHM_FACTOR>
	| <ARITHM_FACTOR> divide <ARITHM_FACTOR>
<ARITHM_FACTOR> ::= id
	| Integer
